\begin{definition}
  A field $F$ is called algebraically closed if every $f\in F[X]$ with $\deg f>0$ has a
  root in $F$.
  \label{<+label+>}
\end{definition}

\begin{theorem}[Fundamental Theorem of Algebra]
  The field $\CC$ is algebraically closed.
  \label{thm:FTAlg}
\end{theorem}
